// Fungsi ini dipanggil apabila butang "Daftar Emel ke ID" ditekan function registerEmail(userEmail) { // 1. Ambil ID yang ditaip oleh pengguna di ruangan input HTML // Pastikan ID input anda dalam HTML adalah "inputID" let idAhli = document.getElementById("inputID").value.trim(); // 2. Validasi: Pastikan ID tidak kosong if (!idAhli) { alert("Sila masukkan ID ahli (contoh: KAF001)!"); return; } // 3. Simpan data ke Firebase // Kita simpan emel di dalam folder ID ahli (contoh: ahli/KAF045/email) // Firebase akan mencipta struktur ahli -> KAF045 -> email secara automatik firebase.database().ref('ahli/' + idAhli).update({ email: userEmail }) .then(() => { // Beritahu pengguna pendaftaran berjaya alert("Emel berjaya didaftarkan!"); // Segarkan halaman supaya nama ahli terus muncul di dashboard location.reload(); }) .catch((error) => { // Tangkap sebarang ralat (contoh: masalah sambungan internet) console.error("Ralat Firebase: ", error); alert("Pendaftaran gagal. Sila semak ID anda."); }); }
